\section{Code Stata de mise en \oe{}uvre - PSM-DD}
\label{ann:B}

Cette annexe présente les principales routines Stata utilisées pour l'estimation
PSM-DD. Le code complet est organisé en scripts numérotés dans le répertoire
\texttt{code/stata/} du projet (un script unique \texttt{tout.do} regroupe
l'ensemble du pipeline). Aucune pondération par poids d'enquête n'est utilisée~;
les erreurs-types sont clusterisées au niveau de la grappe.

\subsection{Statut de traitement stable}

\begin{verbatim}
/* 04_panel.do - traites stables vs jamais traites */
use "$TEMP/traitement_2018.dta", clear
rename D D_2018
merge 1:1 grappe menage using "$TEMP/traitement_2021.dta", ///
    keepusing(D) keep(match) nogenerate
rename D D_2021
gen byte D_stable = .
replace D_stable = 1 if D_2018 == 1 & D_2021 == 1
replace D_stable = 0 if D_2018 == 0 & D_2021 == 0
/* switchers (D_stable == .) exclus du panel causal */
\end{verbatim}

\subsection{Estimation du score de propension (niveau ménage)}

\begin{verbatim}
/* 05_psm_dd.do - Probit menage sur t=0 */
use "$TEMP/panel_vrai.dta", clear
keep if t == 0 & !missing(D) & !missing(log_pcexp)
bysort grappe menage: keep if _n == 1   /* 1 menage = 1 obs */

probit D c.hhsize c.log_pcexp i.milieu i.region ///
         c.hgender c.hage i.heduc i.hmstat, ///
         vce(cluster grappe) nolog
predict pscore, pr
\end{verbatim}

\subsection{Appariement PSM (k-NN) et équilibre}

\begin{verbatim}
/* Appariement k plus proches voisins (k=4, avec remise) */
psmatch2 D, pscore(pscore) neighbor(4) common
pstest hhsize log_pcexp i.milieu i.region hgender hage ///
       i.heduc i.hmstat, both
rename _weight weight_knn
keep grappe menage D pscore weight_knn _support
save "$TEMP/pscore_knn.dta", replace
\end{verbatim}

\subsection{Estimateur PSM-DD}

\begin{verbatim}
/* Poids d'appariement appliques aux enfants du menage */
use "$TEMP/panel_vrai.dta", clear
merge m:1 grappe menage using "$TEMP/pscore_knn.dta", ///
    keepusing(weight_knn) keep(master match) nogenerate
keep if !missing(weight_knn) & weight_knn > 0

foreach outcome in pauvre_AF pauvre_MODA {
    regress `outcome' i.t##i.D [aw=weight_knn], ///
        vce(cluster grappe)
    lincom 1.t#1.D
    di "ATT = " %8.4f r(estimate) ///
       "  SE = " %8.4f r(se) "  p = " %6.4f r(p)
}
\end{verbatim}

\subsection{Effets par dimension N-MODA}

\begin{verbatim}
/* 07_effets_dim.do */
local dims assai eau logem nutri sante protect educ
foreach dim of local dims {
    quietly regress dim_`dim' i.t##i.D [aw=weight_knn], ///
        vce(cluster grappe)
    quietly lincom 1.t#1.D
    di "dim_`dim' : ATT=" %8.4f r(estimate) ///
       "  p=" %6.4f r(p)
}
\end{verbatim}

Le script \texttt{main.do} orchestre l'ensemble du pipeline~: exploration des
bases (\texttt{01\_visitation.do}), variable de traitement
(\texttt{02\_traitement.do}), indicateurs de pauvreté
(\texttt{03\_deprivation.do}), panel et traitement stable
(\texttt{04\_panel.do}), estimation PSM-DD (\texttt{05\_psm\_dd.do}),
statistiques descriptives (\texttt{06\_stats\_desc.do}), effets par dimension
(\texttt{07\_effets\_dim.do}), analyses régionales
(\texttt{08\_carte\_region.do}) et tests de validité
(\texttt{09\_placebo\_attrition.do}).
