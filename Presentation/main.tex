\subsection{Contexte et importance du secteur informel}

\begin{frame}{Contexte général}
  \begin{block}{Poids du secteur informel au Sénégal - ENSIS réalisée en 2011}
    \begin{itemize}
      \item Contribue à plus de \textbf{41,6\,\%} du PIB sénégalais.
      \item Emploie environ \textbf{48,8\,\%} de la main-d'œuvre nationale.
      \item Source principale de revenus pour les ménages vulnérables.
      \item Secteur hétérogène touchant plusieurs branches comme : 
      \pause
    \end{itemize}
    \begin{itemize}
      \item[] \hspace{0.4cm}\small agriculture, commerce, transport ;
      \item[] \hspace{0.4cm}\small manufacture, hébergement, immobilier, etc.
    \end{itemize}
  \end{block}
  \pause
  \begin{alertblock}{Chocs économiques récents}
    \begin{itemize}
      \item COVID-19 : choc sanitaire sans précédent (2020--2021),
      \item Phénomène de l'inflation,
      \item Perturbations géopolitiques et sécheresses récurrentes.
    \end{itemize}
  \end{alertblock}
\end{frame}

\subsection{Problématique et objectifs}

\begin{frame}{Problématique}
  \begin{block}{Question de recherche}
      \justifying
      \large
      \textit{Les effets des chocs économiques sur les branches
      du secteur informel sénégalais sont-ils
      \textbf{transitoires} ou \textbf{permanents} ?}
  \end{block}
  \medskip
  \begin{block}{Chocs transitoires}
    L'économie \textbf{revient} vers sa trajectoire initiale après un choc.
  \end{block}
  \begin{alertblock}{Chocs permanents}
    Le choc \textbf{modifie durablement} le niveau de long terme.
  \end{alertblock}
\end{frame}

\begin{frame}{Objectifs et hypothèses}
  \begin{block}{Objectif général}
    Mesurer la persistance des chocs économiques sur le secteur informel sénégalais.
  \end{block}
  \pause
  \begin{block}{Objectifs spécifiques}
    \begin{itemize}
      \item Tester les propriétés stochastiques des séries sectorielles.
      \item Mesurer la persistance agrégée et désagrégée des chocs.
      \item Évaluer la contribution des chocs issus du secteur formel.
    \end{itemize}
  \end{block}
  \pause
  \begin{block}{Hypothèses de recherche}
    \begin{enumerate}
      \item[H1] Les chocs ont un caractère \textbf{permanent} sur le secteur informel.
      \item[H2] La persistance \textbf{varie} selon les branches.
      \item[H3] Les chocs du secteur formel \textbf{contribuent} à la persistance de l'informel.
    \end{enumerate}
  \end{block}
\end{frame}

\subsection{Modèle VAR multisectoriel}

\begin{frame}[shrink=32]{Modèle VAR multisectoriel}
  \begin{block}{VAR en différences premières}
    \begin{equation}
      \Delta\mathbf{y}_{t} = \mathbf{a} + A_{1}\Delta\mathbf{y}_{t-1} + A_{2}\Delta\mathbf{y}_{t-2}
      + \cdots + A_{p}\Delta\mathbf{y}_{t-p} + \mathbf{u}_{t}
    \end{equation}
    avec
    $\mathbf{y}_{t} \in \mathbb{R}^{7}$ le vecteur du logarithme des valeurs ajoutées informelles,
    et $\Delta\mathbf{y}_{t}=\mathbf{y}_{t}-\mathbf{y}_{t-1}$.
    Ainsi, $\mathbf{a},\mathbf{u}_{t} \in \mathbb{R}^{7}$,
    $A_{j} \in \mathbb{R}^{7\times 7}$ pour $j \in \{1,\ldots,p\}$, et
    $\Sigma_{u}=\mathrm{Var}(\mathbf{u}_{t}) \in \mathbb{S}_{+}^{7}$.
  \end{block}
  \begin{block}{Écriture branche par branche}
    Pour tout $i \in \{1,\ldots,7\}$ :
    \begin{equation}
      \Delta y_{i,t}=a_{i}
      +\sum_{j=1}^{p}\sum_{\ell=1}^{7} a_{i\ell,j}\,\Delta y_{\ell,t-j}
      +u_{i,t}
    \end{equation}
    où $a_{i\ell,j}$ mesure l'effet retardé de la branche $\ell$ sur la branche $i$.
  \end{block}
  \begin{block}{Mesure de la persistance agrégée}
    La matrice de long terme $C(1) \in \mathbb{R}^{7\times 7}$ résume l'effet cumulé des chocs :
    \begin{equation}
      C(1) = \bigl(I_{7} - A_{1} - A_{2} - \cdots - A_{p}\bigr)^{-1}
    \end{equation}
    Pour $i \in \{1,\ldots,7\}$, avec $\mathbf{e}_{i} \in \mathbb{R}^{7}$ le $i$-ième vecteur canonique :
    \begin{equation}
      P_{i} = \mathbf{e}_{i}^{\prime}\,C(1)\,\Sigma_{u}\,C(1)^{\prime}\,\mathbf{e}_{i}
      \in \mathbb{R}_{+}
    \end{equation}
  \end{block}
\end{frame}

\begin{frame}[shrink=18]{Description des trois modèles VAR}
  \begin{block}{Modèle M1 : VAR non restreint}
    Le modèle complet reprend l'écriture branche par branche et comprend $(7p+1)$ paramètres par équation. Pour tout $i \in \{1,\ldots,7\}$ :
    \begin{equation}
      \Delta y_{i,t}=a_i+
      \sum_{j=1}^{p} a_{ii,j}\Delta y_{i,t-j}
      +\sum_{j=1}^{p}\sum_{\substack{1\leq \ell\leq 7\\ \ell\neq i}}
      a_{i\ell,j}\Delta y_{\ell,t-j}+u_{i,t}.
    \end{equation}
  \end{block}
  \begin{block}{Modèle M2 : VAR agrégé restreint}
    Cette version impose $7p(7-2)$ contraintes : $\Delta y_{i,t}$ dépend des retards de la branche $i$ et d'une agrégation des six autres branches.
    \begin{equation}
      \Delta y_{i,t}=a_i+
      \sum_{j=1}^{p} a_{ii,j}\Delta y_{i,t-j}
      +\sum_{j=1}^{p} b_{i,j}\Delta y_{-i,t-j}+u_{i,t},
      \quad \Delta y_{-i,t}=\sum_{\substack{1\leq \ell\leq 7\\ \ell\neq i}}\Delta y_{\ell,t}.
    \end{equation}
  \end{block}
  \begin{alertblock}{Modèle M3 : VAR parcimonieux}
    Le troisième modèle part de M2 et conserve uniquement les coefficients significatifs parmi $a_{ii,j}$ et $b_{i,j}$, afin d'obtenir une spécification plus parcimonieuse.
  \end{alertblock}
\end{frame}

\subsection{Tests et transmission}

\begin{frame}[shrink=25]{Transmission des chocs du secteur formel}
  \begin{block}{Équation de transmission}
    Pour chaque branche $i \in \{1,\ldots,7\}$ :
    \begin{equation}
      \Delta y^{\mathrm{INF}}_{i,t} = \alpha_{i}
      + \sum_{k=1}^{p}\beta_{i,k}\,\Delta y^{\mathrm{INF}}_{i,t-k}
      + \sum_{s=0}^{r}\gamma_{i,s}\,v^{F}_{t-s}
      + \eta_{i,t}
    \end{equation}
    avec $p,r \in \mathbb{N}^{*}$, $y^{\mathrm{INF}}_{i,t},v^{F}_{t},\eta_{i,t} \in \mathbb{R}$,
    et $\alpha_{i},\beta_{i,k},\gamma_{i,s} \in \mathbb{R}$.
    Le terme $v^{F}_{t}$ désigne le choc identifié du secteur formel.
  \end{block}
  \begin{block}{Hypothèses sur les coefficients $\gamma_{i,s}$}
    Pour $i \in \{1,\ldots,7\}$ et $s \in \{0,\ldots,r\}$ :
    \begin{align}
      H_{1,i} &: \gamma_{i,0} = \gamma_{i,1} = \cdots = \gamma_{i,r} = 0
             \quad\text{(pas de transmission)} \notag\\
      H_{2,i} &: \textstyle\sum_{s=0}^{r}\gamma_{i,s} = 0
             \quad\text{(transmission transitoire uniquement)} \notag
    \end{align}
    \begin{itemize}
      \item $H_{1,i}$ non rejetée : \textbf{aucune} transmission significative.
      \item $H_{1,i}$ rejetée, $H_{2,i}$ acceptée : effet \textbf{transitoire}.
      \item $H_{2,i}$ rejetée : effet \textbf{permanent} du choc formel.
    \end{itemize}
  \end{block}
\end{frame}

      Immobilier               & \begin{tabular}[t]{@{}c@{}}1,01\\{\scriptsize(0,17)}\end{tabular} & \begin{tabular}[t]{@{}c@{}}1,02\\{\scriptsize(0,18)}\end{tabular} & \begin{tabular}[t]{@{}c@{}}1,08\\{\scriptsize(0,18)}\end{tabular} & Persistant \\
      Autre                    & \begin{tabular}[t]{@{}c@{}}0,99\\{\scriptsize(0,16)}\end{tabular} & \begin{tabular}[t]{@{}c@{}}1,03\\{\scriptsize(0,18)}\end{tabular} & \begin{tabular}[t]{@{}c@{}}1,01\\{\scriptsize(0,17)}\end{tabular} & Modérément persistant \\

      \textbf{Informel}        & \multicolumn{3}{c}{\begin{tabular}[t]{@{}c@{}}\textbf{1,06}\\{\scriptsize\textbf{(0,19)}}\end{tabular}} & \textbf{Effets durables} \\
      \bottomrule
    \end{tabular*}
  \end{table}
  \vspace{-0.1cm}
  \begin{block}{Lecture}
    Les valeurs entre parenthèses sont les écarts-types. Les résultats confirment une persistance élevée pour l'agriculture, la manufacture et l'immobilier.
  \end{block}
\end{frame}

\subsection{Transmission du secteur formel}

\begin{frame}{Transmission des chocs du secteur formel}
  \begin{block}{Tests de Fisher sur $H_{1}$}
    \begin{itemize}
      \item Aucune branche ne montre de transmission significative au seuil de 5\,\%.
      \item $p$-values comprises entre \textbf{0,255} et \textbf{0,856} selon les branches.
      \item $H_{1}$ (absence de transmission) ne peut être rejetée.
    \end{itemize}
  \end{block}
  \pause
  \begin{alertblock}{Résultat central : segmentation dualiste}
    \begin{itemize}
      \item Le secteur formel \textbf{ne transmet pas} ses chocs vers l'informel.
      \item L'informel sénégalais suit une logique productive largement autonome.
      \item \textbf{H3 non validée} --- segmentation structurelle confirmée.
    \end{itemize}
  \end{alertblock}
  \pause
  \begin{block}{Mécanismes explicatifs}
    Faible intégration des marchés, différence de clientèle,\\
    absorption des chocs par des mécanismes propres à l'informel.
  \end{block}
\end{frame}

\section[Conclusion]{Conclusion et recommandations}
% ============================================================

\subsection{Conclusion générale}

\begin{frame}{Conclusion générale}
  \begin{block}{Principaux résultats}
    \begin{enumerate}
      \item Les chocs économiques laissent des effets \textbf{durables} sur l'informel \\
            (persistance agrégée $= 1{,}06$).
      \pause
      \item Agriculture et manufacture sont les branches \textbf{les plus vulnérables}.
      \pause
      \item Absence de transmission significative du formel :
            \textbf{segmentation dualiste} confirmée.
      \pause
      \item H1 et H2 \textbf{validées} ;
            H3 \textbf{non validée}.
    \end{enumerate}
  \end{block}
\end{frame}

\subsection{Recommandations}

\begin{frame}{Recommandations de politique économique}
  \begin{block}{Soutien aux acteurs informels}
    \begin{itemize}
      \item Cibler les soutiens par branche selon la vulnérabilité.
      \item Développer le micro-crédit et l'assurance-chocs.
      \item Encourager la diversification des revenus.
    \end{itemize}
  \end{block}
  \pause
  \begin{block}{Résilience et formalisation}
    \begin{itemize}
      \item Favoriser une transition formelle graduelle et incitative.
      \item Simplifier la fiscalité pour les petites unités.
      \item Améliorer la production statistique sur l'informel.
    \end{itemize}
  \end{block}
\end{frame}

\subsection{Limites et perspectives}

\begin{frame}[shrink=8]{Limites et perspectives de recherche}
  \begin{alertblock}{Limites de l'étude}
    \begin{itemize}
      \item Reconstitution des données avant 2014 : incertitude statistique.
      \item Seuls les chocs du formel sont testés comme source externe.
      \item Données annuelles : dynamiques infra-annuelles non captées.
      \item Modèle VAR linéaire : pas d'asymétrie de réaction.
    \end{itemize}
  \end{alertblock}
  \pause
  \begin{block}{Contribution}
    Première application de la mesure de persistance multisectorielle à l'économie informelle sénégalaise.
  \end{block}
\end{frame}
